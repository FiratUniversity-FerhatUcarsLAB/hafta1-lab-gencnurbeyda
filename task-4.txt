// Elimizde toplamPasta dilimi var
// Ve toplamKisi sayısı var

FONKSİYON adilPastaPaylasimi(toplamPasta, toplamKisi):

    // Her kişiye düşen dilim sayısı
    dilimKisiBasi = toplamPasta / toplamKisi

    // Kalan dilim varsa onu hesapla
    kalanDilim = toplamPasta % toplamKisi

    YAZDIR "Her kişi " + dilimKisiBasi + " dilim pasta alacak."

    EĞER kalanDilim > 0 ise:
        YAZDIR "Ve " + kalanDilim + " dilim pasta artacak."

    DÖN dilimKisiBasi, kalanDilim
