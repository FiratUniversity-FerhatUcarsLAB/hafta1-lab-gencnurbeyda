// Notları yüksekten düşüğe sıralayan fonksiyon
FONKSİYON notlariSiralaYuksektenAsagiya(ogrenciler):

    // Liste uzunluğunu al
    n = ogrenciler uzunluğu

    // Listenin tüm elemanları üzerinde işlem yap
    TEKRAR n KERE:
        HER i = 0'dan n-2'ye kadar:
            
            // Eğer şu anki öğrencinin notu, yanındakinden küçükse yer değiştir
            EĞER ogrenciler[i].not < ogrenciler[i+1].not:
                TAKAS yap ogrenciler[i] ile ogrenciler[i+1]

    // Sıralanmış listeyi geri döndür
    DÖN ogrenciler
